# Vercel Consolidation & Platform Enhancements — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Consolidate hosting from GitHub Pages to Vercel, add a weekly dead link sweeper cron job, and configure rate limiting on API endpoints.

**Architecture:** Delete the GitHub Actions workflow so Vercel becomes the sole deployment pipeline. Add a cron-triggered serverless function that validates all clip URLs via oEmbed/HEAD, stores results in Vercel Blob, and creates GitHub Issues for dead links. Rate limiting is configured via the Vercel Firewall dashboard (no code).

**Tech Stack:** Vercel serverless functions (Node.js), `@vercel/blob`, GitHub REST API, Vercel Cron Jobs, Vercel Firewall WAF

**Spec:** `docs/superpowers/specs/2026-03-24-vercel-consolidation-design.md`

  type: s.type || "youtube",

// Last synced: ${new Date().toISOString()}
// Total clips: ${clips.length}

export const CLIPS = ${JSON.stringify(clips, null, 2)};
`;

writeFileSync("api/_lib/scenes-data.js", output);
console.log(`Synced ${clips.length} clips to api/_lib/scenes-data.js`);

## Task 3: Video Checker Utility

**Files:**
- Create: `api/_lib/check-video.js`

- [ ] **Step 1: Create the video checker module**

Create `api/_lib/check-video.js`:

```js
/**
 * Builds an oEmbed/HEAD check URL for a clip and validates it.
 * Returns "healthy", "dead", or "unknown".
 */

const OEMBED_URLS = {
  youtube: (id) =>
    `https://www.youtube.com/oembed?url=https://www.youtube.com/watch?v=${id}&format=json`,
  vimeo: (id) =>
    `https://vimeo.com/api/oembed.json?url=https://vimeo.com/${id}`,
  dailymotion: (id) =>
    `https://www.dailymotion.com/services/oembed?url=https://www.dailymotion.com/video/${id}`,
  streamable: (id) =>
    `https://api.streamable.com/oembed.json?url=https://streamable.com/${id}`,
};

/**
 * Check a single clip's liveness.
 * @param {{ id: string, videoId: string|null, type: string, videoUrl: string|null }} clip
 * @returns {Promise<"healthy"|"dead"|"unknown">}
 */
export async function checkClip(clip) {
  try {
    let url;
    let method = "GET";

    if (clip.type === "video" && clip.videoUrl) {
      url = clip.videoUrl;
      method = "HEAD";
    } else if (OEMBED_URLS[clip.type] && clip.videoId) {
      url = OEMBED_URLS[clip.type](clip.videoId);
    } else {
      return "unknown";
    }

    const res = await fetch(url, {
      method,
      signal: AbortSignal.timeout(3000),
    });

    if (res.ok) return "healthy";
    if (res.status === 404 || res.status === 403) return "dead";
    return "unknown";
  } catch {
    return "unknown";
  }
}

/**
 * Check multiple clips with a concurrency limit.
 * @param {Array} clips
 * @param {number} concurrency
 * @returns {Promise<{ healthy: Array, dead: Array, unknown: Array }>}
 */
export async function checkAllClips(clips, concurrency = 50) {
  const results = { healthy: [], dead: [], unknown: [] };
  let index = 0;

  async function worker() {
    while (index < clips.length) {
      const clip = clips[index++];
      const status = await checkClip(clip);
      results[status].push(clip);
    }
  }

  const workers = Array.from(
    { length: Math.min(concurrency, clips.length) },
    () => worker()
  );
  await Promise.all(workers);

  return results;
}
```

- [ ] **Step 2: Create the sweep handler**

Create `api/sweep.js`:

```js
import { put } from "@vercel/blob";
import { CLIPS } from "./_lib/scenes-data.js";
import { checkAllClips } from "./_lib/check-video.js";

const BLOB_KEY = "sweep-report.json";

async function createGitHubIssue(dead) {
  const token = process.env.GITHUB_TOKEN;
  if (!token) return;

  const date = new Date().toISOString().slice(0, 10);
  const rows = dead
    .map((c) => `| ${c.id} | ${c.videoId || c.videoUrl || "—"} | ${c.type} | ${c.source} |`)
    .join("\n");

  const body = `## Dead Links Report — ${date}

| Clip ID | Video ID / URL | Type | Source |
|---------|----------------|------|--------|
${rows}

_Auto-generated by \`/api/sweep\` cron job._`;

  try {
    await fetch("https://api.github.com/repos/alltheusernamesaretakensoistarted/so-i-started-blasting/issues", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${token}`,
        Accept: "application/vnd.github+json",
        "X-GitHub-Api-Version": "2022-11-28",
      },
      body: JSON.stringify({
        title: `🔗 Dead links found: ${dead.length} clips (${date})`,
        body,
        labels: ["dead-links"],
      }),
    });
  } catch {
    // Fire-and-forget — blob report is the primary output
  }
}

export default async function handler(req, res) {
  // Verify cron secret
  const auth = req.headers.authorization;
  if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  try {
    const { healthy, dead, unknown } = await checkAllClips(CLIPS);

    const report = {
      timestamp: new Date().toISOString(),
      total: CLIPS.length,
      healthyCount: healthy.length,
      deadCount: dead.length,
      unknownCount: unknown.length,
      dead: dead.map((c) => ({
        id: c.id,
        videoId: c.videoId,
        type: c.type,
        source: c.source,
      })),
      unknown: unknown.map((c) => ({
        id: c.id,
        videoId: c.videoId,
        type: c.type,
        source: c.source,
      })),
    };

    await put(BLOB_KEY, JSON.stringify(report), {
      access: "public",
      addRandomSuffix: false,
      contentType: "application/json",
    });

    if (dead.length > 0) {
      await createGitHubIssue(dead);
    }

    return res.status(200).json({
      message: `Sweep complete: ${dead.length} dead, ${unknown.length} unknown, ${healthy.length} healthy`,
    });
  } catch (err) {
    return res.status(500).json({ error: err.message });
  }
}
```

## Task 5: Sweep Report Endpoint

**Files:**
- Create: `api/sweep-report.js`

- [ ] **Step 1: Create the report handler**

Create `api/sweep-report.js`:

```js
import { head } from "@vercel/blob";

const BLOB_KEY = "sweep-report.json";

export default async function handler(req, res) {
  if (req.method !== "GET") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  try {
    const blob = await head(BLOB_KEY);
    const response = await fetch(blob.url);
    const report = await response.json();
    return res.status(200).json(report);
  } catch (err) {
    if (err.code === "blob_not_found") {
      return res.status(200).json({ error: "No sweep data yet" });
    }
    return res.status(500).json({ error: err.message });
  }
}
```
